def pop_head
	elim_nodo = @head
	if(!empty)
		@head = elim_nodo.next
		elim_nodo = nil
		@size -= 1
	end

def push_head(value)
	
	if(empty)
		newNode = Node.new(value,nil,nil)
		@head = newNode
		@tail = newNode
	else
		newNode = Node.new(value,@head,nil)
		@head.prev = newNode
		@head = newNode
	end
	@size += 1

def push_tail(value)
	if(empty)
		newNode = Node.new(value,nil,nil)
		@head = newNode
		@tail = newNode
	else
		newNode = Node.new(value,nil,@tail)
		@tail.next = newNode
		@tail = newNode
	end
	@size += 1
end
